$ yum install bind bind-utils -y

#### 一 正向解析

```
浅红色文字：<font color="#dd0000">浅红色文字：</font><br />
# 编辑 /etc/named.conf
options {
	listen-on port 53 { 192.168.0.104; };
	directory 	"/var/named";
	dump-file 	"/var/named/data/cache_dump.db";
	statistics-file "/var/named/data/named_stats.txt";
	memstatistics-file "/var/named/data/named_mem_stats.txt";
	recursing-file  "/var/named/data/named.recursing";
	secroots-file   "/var/named/data/named.secroots";
	allow-query     { any; };
recursion yes;
	dnssec-enable no;
	dnssec-validation no;
        forwarders      { 192.168.0.2; };
	/* Path to ISC DLV key */
	bindkeys-file "/etc/named.root.key";
	managed-keys-directory "/var/named/dynamic";
	pid-file "/run/named/named.pid";
	session-keyfile "/run/named/session.key";
};
logging {
        channel default_debug {
                file "data/named.run";
                severity dynamic;
        };
};
zone "." IN {
	type hint;
	file "named.ca";
};
include "/etc/named.rfc1912.zones";
include "/etc/named.root.key";

检测主配置文件
$ named-checkconf


# 编辑 /etc/named.rfc1912.zones
zone "host.com" IN {
        type  master;
        file  "host.com.zone";
        allow-update { 192.168.0.104; };
};


# 编辑 /var/named/host.com.zone  文件内最下,添加一下内容
$ORIGIN host.com.
$TTL 600	; 10 minutes
@       IN SOA	dns.host.com. dnsadmin.host.com. (
				2019111002 ; serial
				10800      ; refresh (3 hours)
				900        ; retry (15 minutes)
				604800     ; expire (1 week)
				86400      ; minimum (1 day)
				)
			NS   dns.host.com.
$TTL 60	; 1 minute
dns                A    192.168.0.104
k8s-node-01        A    192.168.0.104
k8s-node-02        A    192.168.0.103
k8s-node-03        A    192.168.0.102

检查配置并生效
$ cd /var/named
$ named-checkzone host.com. host.com.zone

bind文件需要注意权限
$ chmod 640 /var/named/host.com.zone 
$ chgrp named /var/named/host.com.zone


添加解析 添加第一条  选择内网 或者  本地DNS解析
$ cat /etc/resolv.conf 
search host.com
nameserver 192.168.0.104
nameserver 100.100.2.138


验证步骤
在k8s-node-01服务器上 都有返回解析 是正确的
$ ping k8s-node-02.host.com
$ ping www.baidu.com

使用dig命令能解析IP地址 是正确的
$ dig -t A k8s-node-03.host.com +short
```
